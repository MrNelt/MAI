exit  - выход из скрипта
echo -n - вывод без перехода на следующую строку

function - описание функции
$(( ... ))  - результат вычисление выражения как в Си
$( ... )  -  результат выполнения функции (команды, утилиты).
${#aaa} длина содеожимого переменной aaa (длина строки).
${aaa:3:4}  - часть строки ааа (интервал) ${строка:от:сколько}.
${aaa:0:4}  - начало строки ааа из 4-х символов
${aаа:$[ ${#aaa} - 4 ]}  - хвост строки ааа из 4-х символов
${@:2:3} - часть списка аргументов
${@:$[ $# - 3 ]}  - последние 4 списка аргументов
arr=(aaaa bbb ccc dd e) - создание массива arr
${arr}  - содержимое 1-го элемента массива arr
${arr[@]}  - содержимое всего массива arr
${arr[2]}  - содержимое 3-го элемента массива arr
${#arr} - длина 1-го элемента массива arr
${#arr[@]} - число элементов массива arr
